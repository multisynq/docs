---
title: "Create Session"
api: "POST https://api.multisynq.io/v1/sessions"
description: "This endpoint creates a new collaboration session."
---

This endpoint creates a new collaboration session for your application.

## Request

<ParamField header="Authorization" type="string" required>
  Bearer token for API authentication
</ParamField>

<ParamField body="appId" type="string" required>
  The application ID to create the session for
</ParamField>

<ParamField body="sessionType" type="string" required>
  The type of session (document, chat, whiteboard, game)
</ParamField>

<ParamField body="initialState" type="object">
  The initial state of the session
</ParamField>

<ParamField body="metadata" type="object">
  Additional session metadata and configuration
</ParamField>

<ParamField body="maxParticipants" type="number">
  Maximum number of participants allowed (default: 10)
</ParamField>

<ParamField body="persistent" type="boolean">
  Whether the session should persist after all users leave (default: false)
</ParamField>

## Response

<ResponseField name="id" type="string">
  The unique identifier of the created session
</ResponseField>

<ResponseField name="appId" type="string">
  The application ID this session belongs to
</ResponseField>

<ResponseField name="sessionType" type="string">
  The type of session created
</ResponseField>

<ResponseField name="state" type="object">
  The initial state of the session
</ResponseField>

<ResponseField name="createdAt" type="string">
  ISO timestamp when the session was created
</ResponseField>

<ResponseField name="metadata" type="object">
  Session metadata and configuration
</ResponseField>

<ResponseField name="wsUrl" type="string">
  WebSocket URL for real-time connection
</ResponseField>
